Hesap Makinesi Uygulaması


          #include < iostream >
          #include < locale.h > // dil kütüphanesi
          #include < math.h > 
          using namespace std;
          int sayi1, sayi2;
          char isaret;
          char devam;
          
          
          int main()
          {
            setlocale(LC_ALL, "Turkish");
            
          
            do
            {
              cout << "Yapmak istediğiniz işlemi seçiniz (+, -, *, /)" << endl;
              cin >> isaret;
              cout << "Lütfen 1. sayıyı giriniz" << endl;
              cin >> sayi1;
              cout << "Lütfen 2, sayiyi giriniz" << endl;
              cin >> sayi2;
            
              switch (isaret)
              {
              case '*':
                cout << "Sonuç = " << sayi1 * sayi2;
                break;
              case '/':
                cout << "Sonuç = " << sayi1 / sayi2;
                break;
              case '+':
                cout << "Sonuç = " << sayi1 + sayi2;
                break;
              case '-':
                cout << "Sonuç = " << sayi1 - sayi2;
                break;
              default:
                break;
              }
          
            } while (true);
          
              cout << "Devam etmek istiyor musun? Evet(y), Hayır(n) :" << endl;
              cin >> devam;
          
              
            
          
            
          
            return 0;
            
          
          
          }
