            #include < iostream >
            #include < string >
            #include < locale.h >
            #include < cstdlib >
            #include < ctime >
            #include < iomanip >
            using namespace std;

            int main() {
                setlocale(LC_ALL, "Turkish");

                int kendiSayim;
                cout << "1 ile 10 arasında bir sayı tutun ve tuttuğunuz sayıyı giriniz :" << endl;
                cin >> kendiSayim;

                srand(static_cast(time(0))); // Rastgele sayı üreteci başlat

                int rastgeleSayi = rand() % 10 + 1; // 1-10 arası rastgele sayı üret
                cout << "Bilgisayarın tuttuğu sayı:" << rastgeleSayi << endl;

                if (rastgeleSayi == kendiSayim) {
                    cout << "Tebrikler kazandınız";
                }

                else {
                    cout << "Üzgünüz kaybettiniz";
                }
            }
